async function shareToXHS({ imageUrl, text }) {
  if (!navigator.clipboard?.write || typeof ClipboardItem === 'undefined') {
    alert('Sharing not supported on this browser');
    return;
  }

  try {
    // --- 1️⃣ Fetch image from proxy ---
    // Proxy should return JPEG; compress/resize to <800KB on backend
    const proxyUrl = `/api/public/image-proxy?url=${encodeURIComponent(imageUrl)}`;
    const imagePromise = fetch(proxyUrl, {
      cache: 'no-store',
      credentials: 'omit'
    }).then(async r => {
      if (!r.ok) throw new Error('Image fetch failed');
      const blob = await r.blob();

      // Optional: log blob size for debugging
      console.log('Blob size (KB):', (blob.size / 1024).toFixed(1));

      // Extra safety: warn if too big
      if (blob.size > 800 * 1024) {
        console.warn('Image exceeds 800KB, may be ignored by XHS');
      }

      return blob;
    });

    // --- 2️⃣ Prepare clipboard item (image + text) ---
    const clipboardItem = new ClipboardItem({
      'image/jpeg': imagePromise,
      'text/plain': new Blob([text], { type: 'text/plain' })
    });

    // --- 3️⃣ Write to clipboard ---
    // ShareLah style: chain .then(), do NOT await
    navigator.clipboard.write([clipboardItem])
      .then(() => {
        // --- 4️⃣ Micro-delay before opening XHS ---
        setTimeout(() => {
          window.location.href = 'xhsdiscover://post_note?ignore_draft=true';
        }, 25); // 20-30ms delay recommended
      })
      .catch(() => {
        // --- 5️⃣ Fallback if clipboard fails ---
        alert('Failed to prepare post. Please try again.');
      });

  } catch (err) {
    console.error('ShareToXHS error:', err);
    alert('Failed to fetch image. Please try again.');
  }
}
