    return;

  // ---- Critical: iOS file-share preflight
  const canShareFiles =
    navigator.canShare &&
    navigator.canShare({ files: [preparedFile] });

  console.log('[CAN SHARE FILES]', canShareFiles);

  if (!canShareFiles) {
    console.warn(
      '[GOLD SHARE] Safari cannot share files — picker unavoidable'

    return;

      .writeText(shareText)
      .then(() => console.log('[CLIPBOARD] writeText fired'))
      .catch(err =>
        console.warn('[CLIPBOARD] writeText failed', err)
      );
  } catch (err) {
    console.warn('[CLIPBOARD] writeText threw', err);
  }

  // ---- Native share (FILES ONLY — ABSOLUTE RULE)
  try {
    console.log('[NATIVE SHARE] invoking navigator.share');

    navigator.share({
      files: [preparedFile]

      // User cancel is normal — do not treat as error
      console.warn('[NATIVE SHARE] rejected', err);
    });

    console.error('[NATIVE SHARE] threw synchronously', err);
  }

async function fetchImageBlob(url) {

async function normalizeImage(blob) {
  const img = await createImageBitmap(blob);

  const canvas = document.createElement('canvas');
  canvas.width = img.width;
  canvas.height = img.height;

  const ctx = canvas.getContext('2d');
  ctx.drawImage(img, 0, 0);

  return new Promise(resolve => {
    canvas.toBlob(
      b => resolve(b),
      'image/jpeg',
      0.85 // critical: Safari-friendly
    );
  });
}

Step 3: Create the File (FINAL FORM)
function createShareFile(blob) {
  return new File([blob], 'share.jpg', {
    type: 'image/jpeg',
    lastModified: Date.now()
  });
}

Step 4: Preflight Safari acceptance
function validateFileForIOS(file) {
  if (!(file instanceof File)) return false;
  if (!file.type.startsWith('image/')) return false;
  if (file.size > 800 * 1024) return false;

  return (
    navigator.canShare &&
    navigator.canShare({ files: [file] })
  );
}

let preparedFile = null;

async function preloadShareImage(url) {
  const rawBlob = await fetchImageBlob(url);

1️⃣ ESLint Rule Implementation
eslint-rules/no-ios-share-regressions.js
/**
 * Blocks iOS Safari native share regressions.
 *
 * Enforces:
 * - No await inside click handlers
 * - navigator.share({ files }) must be files-only
 * - navigator.canShare({ files }) must be checked
 * - No async file prep inside click handlers
 */

module.exports = {
  meta: {
    type: 'problem',
    docs: {
      description: 'Prevent iOS Safari native share regressions',
    },
    messages: {
      noAwaitInClick:
        '❌ Do not use await inside click handlers. This breaks iOS gesture trust.',
      mixedPayload:
        '❌ Mixed payload in navigator.share(). iOS requires files-only payload.',
      missingCanShare:
        '❌ navigator.share() must be gated by navigator.canShare({ files }).',
      asyncFilePrep:
        '❌ Async file preparation is not allowed inside click handlers.',
    },
    schema: [],
  },

  create(context) {
    let insideClickHandler = false;
    let sawCanShareCheck = false;

    function isClickHandler(node) {
      return (
        node.type === 'FunctionExpression' ||
        node.type === 'ArrowFunctionExpression'
      ) &&
      node.parent &&
      (
        node.parent.type === 'AssignmentExpression' ||
        node.parent.type === 'Property'
      ) &&
      /on(click|tap|touchstart)/i.test(
        context.getSourceCode().getText(node.parent)
      );
    }

    return {
      FunctionExpression(node) {
        if (isClickHandler(node)) {
          insideClickHandler = true;
          sawCanShareCheck = false;
        }
      },

      ArrowFunctionExpression(node) {
        if (isClickHandler(node)) {
          insideClickHandler = true;
          sawCanShareCheck = false;
        }
      },

      AwaitExpression(node) {
        if (insideClickHandler) {
          context.report({
            node,
            messageId: 'noAwaitInClick',
          });
        }
      },

      CallExpression(node) {
        const callee = context.getSourceCode().getText(node.callee);

        // Detect navigator.canShare
        if (/navigator\.canShare/.test(callee)) {
          sawCanShareCheck = true;
        }

        // Detect navigator.share
        if (/navigator\.share/.test(callee)) {
          // Must have canShare before
          if (!sawCanShareCheck) {
            context.report({
              node,
              messageId: 'missingCanShare',
            });
          }

          const arg = node.arguments[0];
          if (arg && arg.type === 'ObjectExpression') {
            const keys = arg.properties.map(p => p.key.name);

            if (
              keys.includes('files') &&
              (keys.includes('text') ||
               keys.includes('title') ||
               keys.includes('url'))
            ) {
              context.report({
                node,
                messageId: 'mixedPayload',
              });
            }
          }
        }

        // Async file prep inside click
        if (
          insideClickHandler &&
          /(fetch|blob|arrayBuffer|toBlob|createImageBitmap)/i.test(callee)
        ) {
          context.report({
            node,
            messageId: 'asyncFilePrep',
          });
        }
      },

      'FunctionExpression:exit'() {
        insideClickHandler = false;
        sawCanShareCheck = false;
      },

      'ArrowFunctionExpression:exit'() {
        insideClickHandler = false;
        sawCanShareCheck = false;
      },
    };
  },
};
